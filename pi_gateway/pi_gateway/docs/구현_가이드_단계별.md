```python
# ---------- UART ID (SBC↔STM) ----------
# SBC -> STM32
ID_CMD_VEL   = 0x01   # 이동: float32×3 (vx, vy, wz)
ID_FEED      = 0x05   # 급식: uint8 단계 (0 정지, 1~3)
ID_ARM_START = 0x06   # 로봇팔 동작 시작: uint8 (action_id)
ID_ARM_POSITION_CORRECT = 0x07  # 로봇 위치 보정: float32×3 (dx,dy,dz)
ID_FEED_FILL = 0x08   # 사료 채우기: uint8 (target_percent 0~100)
ID_ARM_WATER = 0x09   # 급수: uint8 (0=정지, 1=버리기, 2=뜨기, 3=전체)
ID_ESTOP     = 0x10   # 비상정지: uint8 (0/1)
```

**함수 추가 (기존 `make_estop_frame` 아래에):**

```python
def make_arm_start_frame(action_id: int) -> bytes:
    """로봇팔 동작 시작. action_id: 0=정지, 1=변 치우기, 2+=기타"""
    payload = bytes([action_id & 0xFF])
    return make_frame(0x06, payload)

def make_arm_position_correct_frame(dx: float, dy: float, dz: float) -> bytes:
    """로봇 위치 보정. dx,dy,dz (m)"""
    payload = struct.pack("<fff", float(dx), float(dy), float(dz))
    return make_frame(0x07, payload)

def make_feed_fill_frame(target_percent: int) -> bytes:
    """사료 채우기. target_percent: 0~100"""
    payload = bytes([min(100, max(0, target_percent)) & 0xFF])
    return make_frame(0x08, payload)

def make_arm_water_frame(water_action: int) -> bytes:
    """급수. water_action: 0=정지, 1=물 버리기, 2=물 다시 뜨기, 3=전체"""
    payload = bytes([water_action & 0xFF])
    return make_frame(0x09, payload)
```

## Phase 2: 고양이 탐지 모듈

### 2.1 새 파일 생성: `src/cat_detection.py`

```python
"""cat_detection.py — USB 웹캠 + AI 모델로 고양이 탐지, 속도 감소."""

import threading
import time
from typing import Optional

try:
    import cv2
except ImportError:
    cv2 = None

# AI 모델은 YOLO, TensorFlow Lite 등 사용 가능
# 예시: YOLO 사용
try:
    from ultralytics import YOLO
    YOLO_AVAILABLE = True
except ImportError:
    YOLO_AVAILABLE = False
    YOLO = None


class CatDetector:
    """USB 웹캠에서 고양이 탐지, 속도 감소 multiplier 제공."""
    
    def __init__(self, model_path: Optional[str] = None, camera_id: int = 0, 
                 speed_reduction_factor: float = 0.5, detection_threshold: float = 0.5):
        """
        Args:
            model_path: AI 모델 파일 경로 (YOLO .pt, TensorFlow Lite .tflite 등)
            camera_id: 웹캠 ID (보통 0)
            speed_reduction_factor: 고양이 감지 시 속도 감소 비율 (0.5 = 50% 감소)
            detection_threshold: 탐지 신뢰도 임계값
        """
        self.model_path = model_path
        self.camera_id = camera_id
        self.speed_reduction_factor = float(speed_reduction_factor)
        self.detection_threshold = float(detection_threshold)
        
        self.cat_detected = False
        self._running = False
        self._detect_thread = None
        self._model = None
        self._camera = None
        
        if model_path and YOLO_AVAILABLE:
            try:
                self._model = YOLO(model_path)
            except Exception as e:
                print(f"[CatDetector] 모델 로드 실패: {e}")
        
    def start(self):
        """탐지 스레드 시작."""
        if cv2 is None:
            print("[CatDetector] cv2 없음, 탐지 비활성화")
            return
        
        if self._model is None:
            print("[CatDetector] 모델 없음, 탐지 비활성화")
            return
        
        try:
            self._camera = cv2.VideoCapture(self.camera_id)
            if not self._camera.isOpened():
                print(f"[CatDetector] 웹캠 {self.camera_id} 열기 실패")
                return
        except Exception as e:
            print(f"[CatDetector] 웹캠 오픈 실패: {e}")
            return
        
        self._running = True
        self._detect_thread = threading.Thread(target=self._detect_loop, daemon=True)
        self._detect_thread.start()
        print("[CatDetector] 시작됨")
    
    def stop(self):
        """탐지 스레드 종료."""
        self._running = False
        if self._camera:
            try:
                self._camera.release()
            except Exception:
                pass
        self._camera = None
    
    def _detect_loop(self):
        """탐지 루프 (스레드에서 실행)."""
        while self._running and self._camera:
            try:
                ret, frame = self._camera.read()
                if not ret:
                    time.sleep(0.1)
                    continue
                
                # 모델 추론
                if self._model:
                    results = self._model(frame, verbose=False)
                    # YOLO 결과 파싱
                    cat_found = False
                    for result in results:
                        boxes = result.boxes
                        for box in boxes:
                            cls = int(box.cls[0])
                            conf = float(box.conf[0])
                            # 클래스 이름이 "cat"이거나 클래스 ID가 고양이인 경우
                            # (모델에 따라 다를 수 있음, COCO는 클래스 15가 고양이)
                            if (cls == 15 or result.names.get(cls, "") == "cat") and conf >= self.detection_threshold:
                                cat_found = True
                                break
                        if cat_found:
                            break
                    self.cat_detected = cat_found
                else:
                    self.cat_detected = False
                
            except Exception as e:
                print(f"[CatDetector] 탐지 오류: {e}")
            
            time.sleep(0.1)  # 10Hz
    
    def get_speed_multiplier(self) -> float:
        """속도 multiplier 반환. 고양이 감지 시 reduction_factor, 아니면 1.0"""
        return self.speed_reduction_factor if self.cat_detected else 1.0
```

```python
# ---------- 로봇팔 제어 브릿지 (DDS → UART) ----------
class ArmControlBridge(Node):
    """젯슨에서 오는 로봇팔 제어 명령을 UART로 전송."""
    
    def __init__(self, uart: UartLink):
        super().__init__("arm_control_bridge")
        self.uart = uart
        
        # DDS 토픽 구독 (젯슨 → 파이)
        self.sub_start = self.create_subscription(
            UInt8, "arm/start", self.on_start, 10
        )
        self.sub_correct = self.create_subscription(
            Vector3, "arm/position_correct", self.on_correct, 10
        )
        self.sub_water = self.create_subscription(
            UInt8, "arm/water", self.on_water, 10
        )
        
        from src.uart_frames import (
            make_arm_start_frame, 
            make_arm_position_correct_frame,
            make_arm_water_frame
        )
        self.make_arm_start = make_arm_start_frame
        self.make_arm_position_correct = make_arm_position_correct_frame
        self.make_arm_water = make_arm_water_frame
    
    def on_start(self, msg: UInt8):
        """arm/start 토픽 수신 → UART 전송"""
        try:
            action_id = int(msg.data)
            frame = self.make_arm_start(action_id)
            self.uart.send(frame)
            self.get_logger().info(f"arm/start: action_id={action_id}")
        except Exception as e:
            self.get_logger().error(f"arm/start 처리 오류: {e}")
    
    def on_correct(self, msg: Vector3):
        """arm/position_correct 토픽 수신 → UART 전송"""
        try:
            dx, dy, dz = float(msg.x), float(msg.y), float(msg.z)
            frame = self.make_arm_position_correct(dx, dy, dz)
            self.uart.send(frame)
            self.get_logger().info(f"arm/position_correct: dx={dx}, dy={dy}, dz={dz}")
        except Exception as e:
            self.get_logger().error(f"arm/position_correct 처리 오류: {e}")
    
    def on_water(self, msg: UInt8):
        """arm/water 토픽 수신 → UART 전송"""
        try:
            water_action = int(msg.data)
            frame = self.make_arm_water(water_action)
            self.uart.send(frame)
            self.get_logger().info(f"arm/water: action={water_action}")
        except Exception as e:
            self.get_logger().error(f"arm/water 처리 오류: {e}")
```

```python
class TeleopPublisher(Node):
    def __init__(self, cfg: TeleopConfig, cat_detector=None):  # cat_detector 추가
        super().__init__("teleop_publisher")
        self.cfg = cfg
        self.cat_detector = cat_detector  # 추가
        self.pub = self.create_publisher(Twist, "cmd_vel_teleop", 10)
        self.timer = self.create_timer(1.0/30.0, self.tick)

    def tick(self):
        msg = Twist()
        if STATE.mode == "teleop" and (not STATE.estop):
            vx, vy, wz = state_to_cmd(STATE, self.cfg)
            # 고양이 탐지 속도 감소 적용
            if self.cat_detector:
                multiplier = self.cat_detector.get_speed_multiplier()
                vx *= multiplier
                vy *= multiplier
                wz *= multiplier  # 회전도 감소할지 결정
            msg.linear.x = vx
            msg.linear.y = vy
            msg.angular.z = wz
        self.pub.publish(msg)
```

```python
    """로봇팔 동작 시작. body: {"action_id": 1}"""
```

```bash
ros2 topic pub /arm/position_correct geometry_msgs/msg/Vector3 "{x: 0.1, y: 0.0, z: 0.0}"
ros2 topic pub /arm/water std_msgs/msg/UInt8 "{data: 3}"
ros2 topic pub /feed/fill std_msgs/msg/UInt8 "{data: 80}"
```

### 4. Web API 테스트

```bash
# 로봇팔 시작
curl -X POST http://localhost:8000/arm/start -H "Content-Type: application/json" -d '{"action_id": 1}'

# 위치 보정
curl -X POST http://localhost:8000/arm/position_correct -H "Content-Type: application/json" -d '{"dx": 0.1, "dy": 0.0, "dz": 0.0}'

# 급수
curl -X POST http://localhost:8000/arm/water -H "Content-Type: application/json" -d '{"action": 3}'

# 사료 채우기
curl -X POST http://localhost:8000/feed/fill -H "Content-Type: application/json" -d '{"target_percent": 80}'
```
