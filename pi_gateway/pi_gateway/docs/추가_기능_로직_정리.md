##### `tick(now: float)`

```python
def tick(self, now: float):
    """패트롤 중 호출. 시간 간격에 따라 액션 실행."""
    # 변치우기 (ID_ARM_START, action_id=1)
    if self.schedule.litter_clean_interval_s > 0:
        if now - self._last_litter_clean_ts >= self.schedule.litter_clean_interval_s:
            try:
                frame = make_arm_start_frame(1)  # action_id=1: 변 치우기
                self.uart.send(frame)
                self._last_litter_clean_ts = now
                print(f"[PatrolAction] Litter clean triggered (interval={self.schedule.litter_clean_interval_s}s)")
            except Exception as e:
                print(f"[PatrolAction] Litter clean failed: {e}")
    
    # 자동 급식 (ID_FEED, level=auto_feed_level)
    if self.schedule.auto_feed_interval_s > 0:
        if now - self._last_auto_feed_ts >= self.schedule.auto_feed_interval_s:
            try:
                level = max(1, min(3, self.schedule.auto_feed_level))  # 1~3 범위
                frame = make_feed_frame(level)
                self.uart.send(frame)
                self._last_auto_feed_ts = now
                print(f"[PatrolAction] Auto feed triggered (level={level}, interval={self.schedule.auto_feed_interval_s}s)")
            except Exception as e:
                print(f"[PatrolAction] Auto feed failed: {e}")
    
    # 급수 (ID_ARM_WATER, action=3: 전체)
    if self.schedule.water_interval_s > 0:
        if now - self._last_water_ts >= self.schedule.water_interval_s:
            try:
                frame = make_arm_water_frame(3)  # action=3: 전체 (버리기 → 다시 뜨기)
                self.uart.send(frame)
                self._last_water_ts = now
                print(f"[PatrolAction] Water triggered (interval={self.schedule.water_interval_s}s)")
            except Exception as e:
                print(f"[PatrolAction] Water failed: {e}")
```

**역할:**
- 패트롤 중 매 프레임 호출
- 각 액션의 시간 간격을 체크하여 실행 시간이 되면 UART로 명령 전송

**로직 흐름:**
1. 변치우기 체크:
   - `litter_clean_interval_s > 0`이고
   - 현재 시간 - 마지막 실행 시간 >= 간격이면
   - `ID_ARM_START` (action_id=1) 프레임 전송

2. 자동 급식 체크:
   - `auto_feed_interval_s > 0`이고
   - 현재 시간 - 마지막 실행 시간 >= 간격이면
   - `ID_FEED` (level=auto_feed_level) 프레임 전송

3. 급수 체크:
   - `water_interval_s > 0`이고
   - 현재 시간 - 마지막 실행 시간 >= 간격이면
   - `ID_ARM_WATER` (action=3) 프레임 전송

**호출 위치:**
- `PatrolLoop.tick()` → 패트롤 모드일 때만 호출

##### `tick()` 수정

```python
def tick(self):
    now = time.time()
    dt = now - self._last_ts
    self._last_ts = now
    out = Twist()
    if self.estop:
        self.pub_auto.publish(out)
        return
    if self.mode == "auto":
        vx, wz = self.profile.step(dt)
        out.linear.x = float(vx)
        out.angular.z = float(wz)
        
        # 패트롤 중 액션 스케줄러 실행
        if self.action_scheduler is not None:
            self.action_scheduler.tick(now)
    
    self.pub_auto.publish(out)
```

**변경 사항:**
- 패트롤 모드(`mode == "auto"`)일 때 `action_scheduler.tick(now)` 호출 추가

```python
@app.post("/patrol/actions/schedule")
async def set_patrol_action_schedule(body: dict):
    """
    패트롤 중 액션 스케줄 설정.
    
    Body:
        litter_clean_interval_s (float, optional): 변치우기 간격 (초). 0이면 비활성화
        auto_feed_interval_s (float, optional): 자동 급식 간격 (초). 0이면 비활성화
        auto_feed_level (int, optional): 자동 급식 레벨 (1~3, 기본 2)
        water_interval_s (float, optional): 급수 간격 (초). 0이면 비활성화
    """
    if _patrol_node is None:
        return JSONResponse(status_code=503, content={"ok": False, "error": "Patrol node not available"})
    
    try:
        from src.ros_cmdvel import PatrolActionSchedule
        
        schedule = PatrolActionSchedule(
            litter_clean_interval_s=float(body.get("litter_clean_interval_s", 0.0)),
            auto_feed_interval_s=float(body.get("auto_feed_interval_s", 0.0)),
            auto_feed_level=int(body.get("auto_feed_level", 2)),
            water_interval_s=float(body.get("water_interval_s", 0.0)),
        )
        
        # 범위 체크
        schedule.auto_feed_level = max(1, min(3, schedule.auto_feed_level))
        
        # PatrolLoop에 스케줄 업데이트
        _patrol_node.update_action_schedule(schedule)
        
        return {
            "ok": True,
            "schedule": {
                "litter_clean_interval_s": schedule.litter_clean_interval_s,
                "auto_feed_interval_s": schedule.auto_feed_interval_s,
                "auto_feed_level": schedule.auto_feed_level,
                "water_interval_s": schedule.water_interval_s,
            }
        }
    except Exception as e:
        return JSONResponse(status_code=500, content={"ok": False, "error": str(e)})
```

```
1. 패트롤 모드 시작 (mode="auto")
   ↓
2. PatrolLoop.tick() 호출 (30Hz)
   ↓
3. PatrolActionScheduler.tick(now) 호출
   ↓
4. 각 액션 시간 간격 체크:
   - 변치우기: now - _last_litter_clean_ts >= interval?
   - 자동 급식: now - _last_auto_feed_ts >= interval?
   - 급수: now - _last_water_ts >= interval?
   ↓
5. 시간이 되면 UART 프레임 전송:
   - make_arm_start_frame(1) → 변치우기
   - make_feed_frame(level) → 자동 급식
   - make_arm_water_frame(3) → 급수
   ↓
6. STM32에서 서보모터 제어
```

```python
def on_connect(client, userdata, flags, *args):
    reason_code = args[0] if args else 0
    if reason_code != 0:
        print(f"[MQTT] 연결 실패: reason_code={reason_code}")
        print(f"[MQTT] 연결 실패 원인: {_get_reason_code_name(reason_code)}")
        return
    print(f"[MQTT] ✅ 연결 성공! 토픽 구독: {TOPIC_CONTROL}")
    client.subscribe(TOPIC_CONTROL)


def _get_reason_code_name(code: int) -> str:
    """MQTT 연결 실패 원인 코드 설명."""
    codes = {
        1: "잘못된 프로토콜 버전",
        2: "클라이언트 ID 거부",
        3: "서버 사용 불가",
        4: "잘못된 사용자명/비밀번호",
        5: "인증 실패",
    }
    return codes.get(code, f"알 수 없는 오류 (코드: {code})")
```

**역할:**

```python
def _status_loop(mqttc: mqtt.Client):
    """상태 발행 루프. 주기적으로 /sub/robot/status 토픽에 발행."""
    publish_count = 0
    last_error_time = 0.0
    
    while True:
        time.sleep(STATUS_INTERVAL)
        payload = _build_status()
        try:
            result = mqttc.publish(TOPIC_STATUS, json.dumps(payload), qos=0)
            if result.rc == mqtt.MQTT_ERR_SUCCESS:
                publish_count += 1
                if publish_count % 50 == 0:  # 50번마다 로그 (약 10초마다)
                    print(f"[MQTT] 상태 발행 중... (발행 횟수: {publish_count})")
            else:
                if time.time() - last_error_time > 5.0:  # 5초마다 에러 로그
                    print(f"[MQTT] 상태 발행 실패: rc={result.rc}")
                    last_error_time = time.time()
        except Exception as e:
            if time.time() - last_error_time > 5.0:
                print(f"[MQTT] 상태 발행 예외: {e}")
                last_error_time = time.time()
```

**역할:**
- 상태 발행 성공 시 주기적 로그 출력 (약 10초마다)
- 상태 발행 실패 시 에러 로그 출력 (5초마다)
- 발행 횟수 추적

**로직 흐름:**
1. `STATUS_INTERVAL` (기본 0.2초)마다 반복
2. `_build_status()`로 상태 데이터 생성
3. MQTT로 발행
4. 성공 시: 50번마다 로그 출력
5. 실패 시: 5초마다 에러 로그 출력
