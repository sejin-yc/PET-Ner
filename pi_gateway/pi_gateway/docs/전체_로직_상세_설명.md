## 1. 수동 제어 (Teleop)

- **주요 함수**: `state_to_cmd()`, `start_demo_teleop_loop()`, `ws_teleop()`

### 🔧 동작 원리

#### 1.1 웹 UI 입력 수신
**파일**: `src/web_ws_server.py`

```python
@app.websocket("/ws/teleop")
async def ws_teleop(ws: WebSocket):
    # WebSocket으로 실시간 입력 수신
    data = await ws.receive_json()
    
    if t == "press":  # 키보드 입력
        key = str(data.get("key", ""))  # "up", "down", "left", "right", "rot_l", "rot_r"
        down = bool(data.get("down", False))
        if down:
            STATE.pressed.add(key)
        else:
            STATE.pressed.discard(key)
    
    elif t == "joy":  # 조이스틱 입력
        STATE.joy_x = float(data.get("joy_x", 0.0))  # 전진(+)/후진(-)
        STATE.joy_y = float(data.get("joy_y", 0.0))  # 좌(+)/우(-)
        STATE.joy_active = bool(data.get("joy_active", False))
```

**데이터 구조**:
- `STATE.pressed`: 키보드 입력 집합 (`{"up", "down", "left", "right", "rot_l", "rot_r"}`)
  - **FE에서는 실제로 4개만 사용**: W(up), S(down), A(left/좌회전), D(right/우회전)
  - `left`, `right`는 **좌회전/우회전** (`wz` 제어)
  - `rot_l`, `rot_r`도 지원하지만 `left`, `right`와 동일하게 동작 (호환성)
- `STATE.joy_x`, `STATE.joy_y`: 조이스틱 좌표 (-1.0 ~ 1.0)
- `STATE.joy_active`: 조이스틱 활성화 여부

#### 1.2 속도 계산
**파일**: `src/main.py:37-61`

```python
def state_to_cmd(state, cfg: TeleopConfig, cat_detector: Optional[CatDetector] = None):
    # 1. 비상정지 체크
    if getattr(state, "estop", False):
        return 0.0, 0.0, 0.0
    
    # 2. 조이스틱 입력 우선 처리
    if getattr(state, "joy_active", False):
        vx = float(getattr(state, "joy_x", 0.0)) * cfg.linear_speed
        vy = float(getattr(state, "joy_y", 0.0)) * cfg.linear_speed
        wz = 0.0  # 조이스틱 회전은 별도 처리 (FE에서 처리)
    else:
        # 3. 키보드 입력 처리
        pressed = getattr(state, "pressed", set()) or set()
        vx = vy = wz = 0.0
        if "up" in pressed:    vx += cfg.linear_speed      # W: 전진
        if "down" in pressed:  vx -= cfg.linear_speed      # S: 후진
        if "left" in pressed:  wz += cfg.angular_speed     # A: 좌회전
        if "right" in pressed: wz -= cfg.angular_speed      # D: 우회전
        if "rot_l" in pressed: wz += cfg.angular_speed     # 좌회전 (rot_l도 지원, left와 동일)
        if "rot_r" in pressed: wz -= cfg.angular_speed     # 우회전 (rot_r도 지원, right와 동일)
    
    # 4. 고양이 탐지 시 속도 감소
    if cat_detector:
        multiplier = cat_detector.get_speed_multiplier()  # 0.5 또는 1.0
        vx *= multiplier
        vy *= multiplier
        wz *= multiplier
    
    return vx, vy, wz
```

**계산 로직**:
- 조이스틱 활성화 시: `vx = joy_x * linear_speed`, `vy = joy_y * linear_speed`
- 키보드 입력 시: 각 키에 따라 속도 누적
- 고양이 탐지 시: 모든 속도에 `speed_reduction_factor` (기본 0.5) 곱하기

#### 2.2 PatrolProfile 패턴 재생
**파일**: `src/ros_cmdvel.py:56-78`

```python
class PatrolProfile:
    def step(self, dt: float) -> Tuple[float, float]:
        self._seg_elapsed += dt  # 경과 시간 누적
        
        seg = self.cfg.segments[self._seg_idx]  # 현재 구간
        
        # 구간 시간이 지나면 다음 구간으로 순환
        while self._seg_elapsed >= seg.duration_s and seg.duration_s > 0:
            self._seg_elapsed -= seg.duration_s
            self._seg_idx = (self._seg_idx + 1) % len(self.cfg.segments)  # 순환
            seg = self.cfg.segments[self._seg_idx]
        
        return seg.vx_mps, seg.wz_rps  # 현재 구간의 속도 반환
```

**기본 패턴** (`src/ros_cmdvel.py:49-52`):

#### 2.3 cmd_vel_auto 발행
**파일**: `src/ros_cmdvel.py:208-225`

```python
def tick(self):
    now = time.time()
    dt = now - self._last_ts
    self._last_ts = now
    
    out = Twist()
    if self.estop:
        self.pub_auto.publish(out)  # 비상정지 시 속도 0
        return
    
    if self.mode == "auto":
        vx, wz = self.profile.step(dt)  # 패턴에서 속도 가져오기
        out.linear.x = float(vx)
        out.angular.z = float(wz)
        
        # 패트롤 중 액션 스케줄러 실행
        if self.action_scheduler is not None:
            self.action_scheduler.tick(now)
    
    self.pub_auto.publish(out)  # ROS 토픽 발행
```

```python
def _detect_loop(self):
    period = 1.0 / max(0.1, self.detection_rate_hz)  # 기본 10Hz
    
    while self._running:
        ret, frame = self.camera.read()  # USB 웹캠에서 프레임 캡처
        
        if self.use_catstate_model:
            # catstate 모델 추론
            frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
            x = self.transform(image=frame_rgb)["image"].unsqueeze(0).to(self.device)
            logits = self.model(x)  # {"action": ..., "emotion": ...}
            
            # action 또는 emotion 중 하나라도 예측되면 고양이 있음
            cat_found = False
            for head_name, head_logits in logits.items():
                probs = torch.nn.functional.softmax(head_logits, dim=-1)
                max_prob = float(probs[0].max().item())
                if max_prob >= self.detection_threshold:
                    cat_found = True
                    break
        else:
            # Hugging Face 모델 추론
            inputs = self.processor(frame_rgb, return_tensors="pt").to(self.device)
            with torch.no_grad():
                outputs = self.model(**inputs)
                probs = torch.nn.functional.softmax(outputs.logits, dim=-1)
                cat_prob = probs[0][self.cat_class_id].item()
                cat_found = (cat_prob >= self.detection_threshold)
        
        with self._lock:
            self.cat_detected = cat_found
        
        time.sleep(period)
```

```python
def make_arm_start_frame(action_id: int) -> bytes:
    """
    action_id:
    - 0 = 로봇팔 정지 (변 치우기 완료)
    - 1 = 변 치우기 시작 (젯슨의 모방학습 로봇팔이 자율로 4구역 처리)
    """
    payload = bytes([action_id & 0xFF])  # 1 byte
    return make_frame(ID_ARM_START, payload)
```

```python
def on_water(self, msg: UInt8):
    water_action = int(msg.data)
    # 물 버리기(2), 물 받기(3), 물그릇 두기(4)만 바퀴 잠금
    is_wheel_lock = (water_action in [2, 3, 4])

    self.uart.send(frame)
    
    if is_wheel_lock:
        self.arm_water_active = True
        if self.cmdvel_bridge is not None:
            self.cmdvel_bridge.set_arm_water_active(True)

        if self.cmdvel_bridge is not None:
            self.cmdvel_bridge.set_arm_water_active(False)
```

- `2` = 물 버리기 (바퀴 잠금)
- `3` = 물 받기 (바퀴 잠금)

```python
def make_feed_fill_frame(target_percent: int) -> bytes:
    """
    젯슨 AI 모델이 사료통의 빈 양을 계산하여 target_percent를 결정하고,
    STM32의 서보모터가 개폐통을 열어서 급식합니다.
```

```python
def check_litter_clean_on_patrol_start(self, now: float) -> bool:
    import datetime
    
    if self.schedule.litter_clean_daily_hour < 0:
        return False  # 비활성화
    
    if self._litter_clean_executed_this_patrol:
        return False  # 이미 이번 순찰에서 실행함
    
    current_time = datetime.datetime.fromtimestamp(now)
    current_date = current_time.date().isoformat()  # YYYY-MM-DD
    target_hour = self.schedule.litter_clean_daily_hour
    target_minute = self.schedule.litter_clean_daily_minute
    
    # 오늘 아직 변 치우기를 안 했고, 현재 시간이 저녁 시간대 이후인 경우
    if self._last_litter_clean_date != current_date:
        target_time = current_time.replace(hour=target_hour, minute=target_minute, second=0, microsecond=0)
        
        if current_time >= target_time:
            frame = make_arm_start_frame(1)  # action_id=1: 변 치우기
            self.uart.send(frame)
            self._last_litter_clean_date = current_date
            self._litter_clean_executed_this_patrol = True
            return True
    
    return False
```
