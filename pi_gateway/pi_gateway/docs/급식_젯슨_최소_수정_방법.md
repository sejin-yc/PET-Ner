### 3단계: FeedAINode 클래스 추가

```python
class FeedAINode(Node):
    def __init__(self):
        super().__init__("feed_ai_node")
        self.sub_feed_request = self.create_subscription(
            String, "feed/request", self.on_feed_request, 10
        )
        self.pub_feed_amount = self.create_publisher(UInt8, "feed/amount", 10)
        self.get_logger().info("FeedAINode initialized")
    
    def on_feed_request(self, msg: String):
        """Pi Gateway가 급식 요청 시 최신 추론 결과 발행"""
        try:
            import json
            data = json.loads(msg.data)
            waypoint_id = data.get("waypoint_id", "")
            self.get_logger().info(f"Feed requested for waypoint: {waypoint_id}")
            
            # 기존 infer_loop가 계산한 최신 결과 사용
            with FEED_LEVEL_LOCK:
                feed_level = LATEST_FEED_LEVEL if LATEST_FEED_LEVEL is not None else 2
            
            msg = UInt8()
            msg.data = feed_level
            self.pub_feed_amount.publish(msg)
            self.get_logger().info(f"Published feed/amount: {feed_level}")
        except Exception as e:
            self.get_logger().error(f"Failed to process feed request: {e}")
```
